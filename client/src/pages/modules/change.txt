    <script>
        const denominations = [200, 100, 50, 20, 10, 5, 2, 1, 0.50, 0.20, 0.10, 0.05];
        
        // Initialize page
        document.getElementById('voucherDate').valueAsDate = new Date();
        
        // Create denomination inputs
        function createDenominations() {
            const grid = document.getElementById('denomGrid');
            denominations.forEach((denom, idx) => {
                const row = document.createElement('div');
                row.className = 'denom-row';
                row.innerHTML = `
                    <div class="denom-label">GHS ${denom.toFixed(2)}</div>
                    <input type="number" class="denom-input" id="denom${idx}" value="0" min="0" onchange="calculateDenominations()">
                    <div class="denom-total" id="total${idx}">0.00</div>
                `;
                grid.appendChild(row);
            });
        }

        createDenominations();

        // Calculate denomination totals
        function calculateDenominations() {
            let grandTotal = 0;
            denominations.forEach((denom, idx) => {
                const count = parseFloat(document.getElementById(`denom${idx}`).value) || 0;
                const total = denom * count;
                document.getElementById(`total${idx}`).textContent = total.toFixed(2);
                grandTotal += total;
            });
            document.getElementById('cashTotal').textContent = grandTotal.toFixed(2);
        }

        // Sample transaction data
        function populateTransactions() {
            const transactions = [
                { account: 'Covid 1% On Sales', credit: 3.85, debit: 0 },
                { account: 'Sales Account', credit: 384.62, debit: 0 },
                { account: 'Vat On Sales 3%', credit: 11.54, debit: 0 },
                { account: 'Cash Account Ghs', credit: 0, debit: 400.00 },
                { account: 'ROUND OFF', credit: 0, debit: 0.01 }
            ];

            const tbody = document.getElementById('tableBody');
            tbody.innerHTML = '';
            
            let totalCredit = 0;
            let totalDebit = 0;

            transactions.forEach((trans, idx) => {
                totalCredit += trans.credit;
                totalDebit += trans.debit;
                
                const row = document.createElement('tr');
                row.innerHTML = `
                    <td>${idx + 1}</td>
                    <td>${trans.account}</td>
                    <td class="text-right">${trans.credit > 0 ? trans.credit.toFixed(2) : ''}</td>
                    <td class="text-right">${trans.debit > 0 ? trans.debit.toFixed(2) : ''}</td>
                `;
                tbody.appendChild(row);
            });

            // Add total row
            const totalRow = document.createElement('tr');
            totalRow.className = 'total-row';
            totalRow.innerHTML = `
                <td></td>
                <td></td>
                <td class="text-right">${totalCredit.toFixed(2)}</td>
                <td class="text-right">${totalDebit.toFixed(2)}</td>
            `;
            tbody.appendChild(totalRow);
        }

        // Search functionality
        function searchTable() {
            const input = document.getElementById('searchInput').value.toLowerCase();
            const rows = document.querySelectorAll('#tableBody tr:not(.total-row)');
            
            rows.forEach(row => {
                const text = row.textContent.toLowerCase();
                row.style.display = text.includes(input) ? '' : 'none';
            });
        }

        // Change rows display
        function changeRows() {
            const value = document.getElementById('rowsSelect').value;
            console.log(`Displaying ${value} rows`);
        }

        // Post to finance
        function postToFinance() {
            const cashTotal = document.getElementById('cashTotal').textContent;
            const voucherNo = document.getElementById('voucherNo').value || 'Auto-generated';
            const voucherDate = document.getElementById('voucherDate').value;
            const till = document.getElementById('till').value;
            const warehouse = document.getElementById('warehouse').value;

            if (confirm(`Post transaction to finance?\n\nVoucher: ${voucherNo}\nDate: ${voucherDate}\nTill: ${till}\nWarehouse: ${warehouse}\nCash Total: GHS ${cashTotal}`)) {
                alert('Transaction posted to finance successfully!');
                
                // Reset form
                document.getElementById('voucherNo').value = '';
                document.getElementById('tableBody').innerHTML = '<tr><td colspan="4" style="text-align: center; padding: 40px; color: #999;">Click "POPULATE DATA" to load transactions</td></tr>';
                
                // Reset denominations
                denominations.forEach((_, idx) => {
                    document.getElementById(`denom${idx}`).value = '0';
                    document.getElementById(`total${idx}`).textContent = '0.00';
                });
                document.getElementById('cashTotal').textContent = '0.00';
            }
        }
